@composition.pattern(channel=0, length=4) # ← channel=0 = MIDI ch 1 def lsys(p): def expand(s, rules, n): for _ in range(n): s = ''.join(rules.get(c,c) for c in s) return s seq = expand('F', {'F':'F+F-F-F+F'}, 3) beat = 0 for c in seq[:32]: if c == 'F': p.note(60, beat=beat*0.25); beat += 1 elif c == '+': beat += 1
# Sieve of Eratosthenes — prime positions as hits

# 1D Cellular automaton — Rule 30 / 110
@composition.pattern(channel=0, length=4) # ← channel=0 = MIDI ch 1 def cellular(p): def ca(rule_num, width=16): rule = [(rule_num >> i) & 1 for i in range(8)] row = [0]*width; row[width//2] = 1 for _ in range(4): row = [rule[(row[(i-1)%width]<<2)|(row[i]<<1)|row[(i+1)%width]] for i in range(width)] return row for i,v in enumerate(ca(30)): if v: p.note(60+i, beat=i*0.25)

# Reaction-diffusion — Gray-Scott (Turing patterns as velocity)
@composition.pattern(channel=0, length=4) # ← channel=0 = MIDI ch 1 def gray_scott_pat(p): n,f,k = 16,0.055,0.062 u,v = [1.0]*n,[0.0]*n for i in range(n//4, n//4+2): v[i] = 1.0 for _ in range(200): u2=u[:] for i in range(n): uvv=u[i]*v[i]*v[i] u2[i]=u[i]+0.2*(u[(i-1)%n]-2*u[i]+u[(i+1)%n])-uvv+f*(1-u[i]) v[i] =v[i]+0.1*(v[(i-1)%n]-2*v[i]+v[(i+1)%n])+uvv-(f+k)*v[i] u=u2 lo,hi=min(v),max(v) for i,vel in enumerate(v): p.note(60, beat=i*0.25, velocity=int(40+80*(vel-lo)/(hi-lo)))
